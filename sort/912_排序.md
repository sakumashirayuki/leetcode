## 归并排序
### 复杂度分析
时间复杂度
* 最好： $O(n)$
* 最坏： $O(nlog_{2}n)$
* 平均： $O(nlog_{2}n)$<p>

空间复杂度： $O(n)$<p>
稳定性： 稳定<p>
### 源码
```javascript
var sortArray = function(nums) {
    let temp = new Array(nums.length);
    const divide = function(nums, temp, left, right){ // void
        if(left>=right)
            return
        let mid = ((right - left)>>1)+ left;
        divide(nums, temp, left, mid);
        divide(nums, temp, mid + 1, right);
        merge(nums, temp, left, mid, right);
    }
    const merge = function(nums, temp, left, mid, right){
        let i = left;
        let j = mid + 1;
        let t = left; // start pos of temp
        while(i <= mid && j<=right){
            if(nums[i]<=nums[j]){
                temp[t++] = nums[i++];
            }else{
                temp[t++] = nums[j++];
            }
        }
        while(i <= mid){
            temp[t++] = nums[i++];
        }
        while(j <= right){
            temp[t++] = nums[j++];
        }
        for(t = left; t <= right; t++){
            nums[t] = temp[t];
        }
    }
    divide(nums, temp, 0, nums.length - 1);
    return nums;
};
```
## 快速排序
### 复杂度分析
时间复杂度
* 最好： $O(nlog_{2}n)$
* 最坏： $O(n^{2})$
* 平均： $O(nlog_{2}n)$<p>

空间复杂度： $O(nlog_{2}n)$<p>
稳定性： 不稳定<p>
### 源码
```javascript
var sortArray = function(nums) {
    if (nums.length < 2) return nums;
    return  quickSort(nums, 0, nums.length - 1);
};

function quickSort(nums, left, right) {
    if (left >= right) return;
    let pivotIndex = partition(nums, left, right);
    quickSort(nums, left, pivotIndex - 1);
    quickSort(nums, pivotIndex + 1, right);
    return nums;
}

function partition (nums, left, right) { //其实就是双指针把比基准大的数放在后面，比基准小的数放在前面
    let i  = left;
    let j = right;
    let pivot  = nums[left];
    while(i<j){
        while(nums[j]>pivot&&i<j) // j should >= 0
            j--;
        if(i<j)
            nums[i] = nums[j];
        while(nums[i]<pivot&&i<j) // i should < n
            i++;
        if(i<j)
            nums[j] = nums[i];
    }
    nums[i] = pivot; // i equals j
    // i is the position of current pivot
    return i;
}
```
